        // Toast notification
        function showToast(message, type = 'success') {
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.textContent = message;
            document.body.appendChild(toast);
            
            setTimeout(() => {
                toast.style.animation = 'slideIn 0.3s ease-out reverse';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        // Notification Form
        document.getElementById('notificationForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const name = document.getElementById('notifName').value;
            const email = document.getElementById('notifEmail').value;
            
            if (!name || !email) {
                showToast('Per favore, inserisci nome e email', 'error');
                return;
            }
            
            const subscribers = JSON.parse(localStorage.getItem('subscribers') || '[]');
            subscribers.push({ name, email, date: new Date().toISOString() });
            localStorage.setItem('subscribers', JSON.stringify(subscribers));
            
            showToast('Iscrizione completata! Riceverai notifiche sui nuovi episodi.');
            this.reset();
        });

        // Question Form
